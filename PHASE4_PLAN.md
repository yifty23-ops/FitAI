`compute_mesocycle_length(profile, tier)`:
```python
def compute_mesocycle_length(profile, tier):
    if tier == "free":
        return 4  # short, simple
    base = {"beginner": 6, "intermediate": 8, "advanced": 10}[profile.experience]
    if tier == "elite":
        base = min(base + 4, 16)  # elite gets longer cycles
    return base
```

## 4B: POST /plan/generate endpoint

Create `backend/routes/plan.py`:

```python
@router.post("/plan/generate")
async def generate_plan_route(user = Depends(get_current_user), db = Depends(get_db)):
    # 1. Check plan limit
    if not check_plan_limit(user, db):
        raise HTTPException(429, f"Plan limit reached for {user.tier} tier (1/month). Upgrade for unlimited.")

    # 2. Get profile
    profile = db.query(Profile).filter_by(user_id=user.id).first()
    if not profile:
        raise HTTPException(400, "Complete onboarding first")

    # 3. Research (tier-aware, cached)
    research = await research_for_profile(user, profile, db)

    # 4. Generate plan (tier-aware persona)
    result = await generate_plan_tool(user, profile, research, db)

    # 5. Build persona string for storage
    if user.tier == "elite" and user.sport:
        persona_used = f"Elite {user.sport.title()} Coach"
    else:
        persona_used = {"free": "Personal Trainer", "pro": "Pro S&C Coach"}[user.tier]

    # 6. Save
    plan = Plan(
        user_id=user.id,
        tier_at_creation=user.tier,
        profile_snapshot=profile_to_dict(profile),
        mesocycle_weeks=result["plan"]["mesocycle_weeks"],
        plan_data=result["plan"],
        nutrition=result["nutrition"],
        persona_used=persona_used,
        is_active=False
    )
    db.add(plan)
    db.commit()
    db.refresh(plan)

    return {"plan_id": str(plan.id), "plan": result["plan"], "nutrition": result["nutrition"], "persona_used": persona_used}
```
